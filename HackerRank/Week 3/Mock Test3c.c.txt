int compare_numeric_strings(const void *a,const void *b){
    const char *s1=*(const char**)a;
    const char *s2=*(const char**)b;
    size_t len1=strlen(s1);
    size_t len2=strlen(s2);
    if(len1!=len2)
    {
        return (len1>len2)-(len1<len2);
            }
 return strcmp(s1,s2);    
 }
